              <script>
                var councilMembers = ['Thomas Cromwell', 'Thomas Boleyn', 'William Butts, Physician to the King', 'Will Sommers, Court Jester', 'Thomas Cranmer, Archbishop of Canterbury', 'Sir William Compton', 'Sir Thomas More', 'Cardinal Wolsey', 'Charles Brandon, Duke of Suffolk', 'Thomas Howard, Duke of Norfolk', 'Sir Anthony Knyvett'];
                var friends = ['Elliott Turner', 'Brendan Myers', 'Thomas Priestley', 'Rahul Patel', 'Wagner Loor', 'Trevor Warth', 'Stephan Andre', 'Paul Drake', 'Joe Schewtschuk', 'Stefan Pintea', 'Christopher Knudsen'];

                function pair() {
                  var pairings = document.getElementById('council');
                  council.innerHTML = ''; // Clear any existing pairings

                  while (councilMembers.length > 0) {
                    var memberIndex = Math.floor(Math.random() * councilMembers.length);
                    var friendIndex = Math.floor(Math.random() * friends.length);
                    var member = councilMembers[memberIndex];
                    var friend = friends[friendIndex];
                    councilMembers.splice(memberIndex, 1);
                    friends.splice(friendIndex, 1);
                    var li = document.createElement('li');
                    li.textContent = member + ': ' + friend;
                    council.appendChild(li);
                  }
                };

                function reset() {
                  councilMembers = ['Thomas Cromwell', 'Thomas Boleyn', 'William Butts, Physician to the King', 'Will Sommers, Court Jester', 'Thomas Cranmer, Archbishop of Canterbury', 'Sir William Compton', 'Sir Thomas More', 'Cardinal Wolsey', 'Charles Brandon, Duke of Suffolk', 'Thomas Howard, Duke of Norfolk', 'Sir Anthony Knyvett'];
                  friends = ['Elliott Turner', 'Brendan Myers', 'Thomas Priestley', 'Rahul Patel', 'Wagner Loor', 'Trevor Warth', 'Stephan Andre', 'Paul Drake', 'Joe Schewtschuk', 'Stefan Pintea', 'Christopher Knudsen'];
                  document.getElementById('council').innerHTML = '';
                };

                function toggleUL() {
                  let ul = document.getElementById("links");
                  if (ul.style.display === "none") {
                    ul.style.display = "block";
                    ul.style.textAlign = "center";
                    ul.style.margin = "0";
                  } else {
                    ul.style.display = "none";
                  }
                };

              </script>
